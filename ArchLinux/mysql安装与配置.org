** 安装
#+BEGIN_SRC 
$ sudo pacman -S mysql

// 关建的输出： root@localhost: XXXXXXX
// 用户名：root, 密码：XXXXXXX
$ sudo mysqld --initialize --user=mysql --basedir=/usr --datadir=/var/lib/mysql 

$ sudo systemctl enable  mysqld.service
$ sudo systemctl start  mysqld.service
$ mysql -uroot -p
#+END_SRC

** 配置
#+BEGIN_SRC 
$ emacs /etc/mysql/my.cnf
#+END_SRC
